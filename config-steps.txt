1 - Install NVM on:
	https://github.com/coreybutler/nvm-windows

2 - Install NodeJS 10.13

3 - Changed to Nodejs 10.13 in NVM with command:
	nvm use 10.13

4 - Install Yarn:
	npm install -g yarn

5 - Install Next-App
	yarn create next-app

6 - Install Typescript and React/Node Typescript
	 yarn add --dev typescript @types/react @types/node
	
7 - Install tsconfig.json
	touch tsconfig.json

8 - Change stric mode to "true" in tsconfig.json

9 - Create src folder and move folder 'pages' to it

10 - Create .editorconfig

11 - Add .editorconfig settings
	# editorconfig.org
	root = true

	[*]
	indent_style = spaces
	indent_size = 2
	end_of_line = lf
	charset = utf-8
	trim_trailing_whitespace = true
	insert_final_newline = true

12 - Add ESLint plugin in vscode

13 - Add ESLint to project
	npx eslint --init
	Choose this option:
	- To check syntax and find problems
	
14 - Install ESlint plugin in project
	yarn add --dev eslint-plugin-react@latest @typescript-eslint/eslint-plugin@latest @typescript-eslint/parser@latest eslint@latest

15 - Add ESlint React Hooks - https://www.npmjs.com/package/eslint-plugin-react-hooks
	yarn add eslint-plugin-react-hooks --dev

16 - Add React Hooks configuration to ESLint
	{
		"plugins": [
			// ...
			"react-hooks"
		],
		"rules": {
			// ...
			"react-hooks/rules-of-hooks": "error",
			"react-hooks/exhaustive-deps": "warn"
		}
	}

17 - Remove react/prop-types rules from ESLint
	 "rules": {
      	// ...
        "react/prop-types": "off",
    }

18 - Remove react/react-in-jsx-scope rule from ESLint
	"rules": {
      	// ...
        "react/react-in-jsx-scope": "off",
    }

19 - Remove explict-module-boundary rule from ESlint
	"rules": {
      	// ...
        "@typescript-eslint/explicit-module-boundary-types": "off",
    }

20 - Detect React version in ESLint. Add this to .eslintrc.json
	  "settings": {
        "react":{
            "version": "detect"
        }
    },


21 - Configure Prettier
	yarn add --dev eslint-config-prettier eslint-plugin-prettier


22 - Add Husky e lint-staged
	 yarn add --dev lint-staged husky
